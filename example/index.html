public static function danger404($msg = '')
{
  header('HTTP/1.0 404 Not Found');
  if(!$msg) $msg = '404. Страница не найдена';
  self::_()->title .= $msg;
  self::_()->render('site/danger404', array('msg'=>$msg));
  exit;
}
