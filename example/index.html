public static function error404($msg = '')
{
  header('HTTP/1.0 404 Not Found');
  if(!$msg) $msg = '404. Страница не найдена';
  self::_()->title .= $msg;
  self::_()->render('site/error404', array('msg'=>$msg));
  exit;
}
